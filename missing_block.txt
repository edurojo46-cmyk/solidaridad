// Search / filter
function filterComedoresGrid(query) {
    var rawQ = query.trim();
    var grid = document.getElementById('Comedores-grid');
    var results = document.getElementById('Comedores-search-results');
    if (!rawQ) {
        grid.style.display = '';
        results.style.display = 'none';
        results.innerHTML = '';
        return;
    }
    grid.style.display = 'none';
    results.style.display = '';
    
    var normalizeText = function(str) {
        return (str || '').normalize('NFD').replace(/[\u0300-\u036f]/g, '').toLowerCase();
    };
    
    var words = normalizeText(rawQ).split(/\s+/);
    var html = '';
    var count = 0;
    
    zonasData.forEach(function(z) {
        var zNameNorm = normalizeText(z.nombre);
        var zSubNorm = normalizeText(z.sub);
        z.Comedores.forEach(function(ig) {
            var nNorm = normalizeText(ig.n);
            var dNorm = normalizeText(ig.d);
            var hNorm = normalizeText(ig.h);
            
            var searchableText = nNorm + " " + dNorm + " " + hNorm + " " + zNameNorm + " " + zSubNorm;
            var isMatch = words.every(function(w) { return searchableText.indexOf(w) !== -1; });
            
            if (isMatch) {
                count++;
                html += '<div class="ig-detail-item glass" style="animation-delay:'+(count*0.03)+'s">' +
                    '<div class="ig-detail-dot" style="background:'+z.color+'"></div>' +
                    '<div class="ig-detail-info">' +
                        '<h4>'+ig.n+'</h4>' +
                        '<span class="ig-detail-dir"><i class="ri-map-pin-fill"></i> '+ig.d+'</span>' +
                        '<span class="ig-detail-hora"><i class="ri-time-fill"></i> '+ig.h+'</span>' +
                        '<span class="ig-search-zone" style="color:'+z.color+'"><i class="ri-building-4-fill"></i> '+z.nombre+'</span>' +
                    '</div>' +
                '</div>';
            }
        });
    });
    if (count === 0) {
        html = '<div class="ig-no-results"><i class="ri-search-line"></i><p>No se encontraron Comedores con "'+query+'"</p></div>';
    } else {
        html = '<div class="ig-results-count">'+count+' resultado'+(count!==1?'s':'')+'</div>' + html;
    }
    results.innerHTML = html;
}

function scInitMap() {
    if (scMap) { scMap.invalidateSize(); return; }
    scMap = L.map('sc-map', { zoomControl: false }).setView([-34.6037, -58.3816], 13);
    L.tileLayer('https://{s}.basemaps.cartocdn.com/rastertiles/voyager/{z}/{x}/{y}{r}.png', {
        attribution: 'Ã‚Â© OpenStreetMap'
    }).addTo(scMap);
    L.control.zoom({ position: 'bottomright' }).addTo(scMap);
    scLoadAlertas();
    
    if (typeof sbClient !== 'undefined' && sbClient) {
        sbClient.channel('sc-alertas-realtime')
            .on('postgres_changes', { event: '*', schema: 'public', table: 'alertas_calle' }, function() { scLoadAlertas(); })
            .subscribe();
            
        sbClient.channel('sc-mensajes-realtime')
            .on('postgres_changes', { event: '*', schema: 'public', table: 'ayudas_alerta' }, function(payload) {
                var aId = payload.new.alerta_id;
                if (aId) scLoadMensajes(aId);
            })
            .subscribe();
    }
}

async function scLoadMensajes(alertaId) {
    if (typeof sbClient === 'undefined' || !sbClient) return;
    try {
        var { data } = await sbClient.from('ayudas_alerta').select('*').eq('alerta_id', alertaId).order('created_at', { ascending: true });
        if (data) {
            scMensajes[alertaId] = data;
            var c = document.getElementById('sc-mensajes-' + alertaId);
            if (c) c.innerHTML = scRenderMensajes(alertaId);
            var nBox = document.getElementById('sc-necesidades-box-' + alertaId);
            if (nBox) nBox.innerHTML = scRenderNecesidadesBox(alertaId);
        }
    } catch(e) {}
}

    var necIcons = { 'Ropa':'ri-t-shirt-line', 'Abrigo':'ri-windy-line', 'Manta':'ri-hotel-bed-line',

async function scSubmitPledge(alertaId, necesidad) {
    if (typeof auth !== 'undefined' && auth && !auth.isAuthenticated()) {
        if (confirm('Necesitas iniciar sesiÃƒÂ³n para comprometerte a ayudar. Ã‚Â¿Ir al login?')) app.navigate('screen-login');
        return;
    }
    
    var u = auth.getCurrentUser();
    var isValidUUID = u && u.id && /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(u.id);
    var name = u ? (u.user_metadata && u.user_metadata.full_name ? u.user_metadata.full_name : u.email.split('@')[0]) : 'AnÃƒÂ³nimo';
    var text = "Se comprometiÃƒÂ³ a llevar: " + necesidad;
    
    var payload = { alerta_id: alertaId, mensaje: text, helper_name: name, user_id: isValidUUID ? u.id : null, necesidad: necesidad };
    
    // Optimistic UI
    if (!scMensajes[alertaId]) scMensajes[alertaId] = [];
    scMensajes[alertaId].push(payload);
    document.getElementById('sc-mensajes-' + alertaId).innerHTML = scRenderMensajes(alertaId);

    if (nBox) nBox.innerHTML = scRenderNecesidadesBox(alertaId);
    
    try { await sbClient.from('ayudas_alerta').insert([payload]); } 
    catch(e) { scLoadMensajes(alertaId); }
}

function scGeocodeDebouncedForm() {
    clearTimeout(scGeoTimer);
    var loader = document.getElementById('sc-geo-loader');
    if (loader) loader.style.display = 'block';
    scGeoTimer = setTimeout(async function() {
        var pais = document.getElementById('sc-pais').value;
        var ciudad = document.getElementById('sc-ciudad').value.trim();
        var dir = document.getElementById('sc-direccion').value.trim();
        var q = dir + (ciudad ? ', ' + ciudad : '') + ', ' + pais;
        
        if (dir.length < 5) { if (loader) loader.style.display = 'none'; return; }
        try {
            var r = await fetch('https://nominatim.openstreetmap.org/search?format=json&q=' + encodeURIComponent(q) + '&limit=1');
            var d = await r.json();
            if (d && d.length > 0) {
                scGeoLat = parseFloat(d[0].lat);
                scGeoLng = parseFloat(d[0].lon);
                if (isNaN(scGeoLat)) scGeoLat = -34.6037;
                if (isNaN(scGeoLng)) scGeoLng = -58.3816;
                scShowMiniMap(scGeoLat, scGeoLng);
            } else {
                // Si Nominatim no lo encuentra, mostramos Buenos Aires como fallback
                if (scGeoLat === null || isNaN(scGeoLat)) {
                    scGeoLat = -34.6037;
                    scGeoLng = -58.3816;
                    scShowMiniMap(scGeoLat, scGeoLng);
                }
            }
        } catch(e) {
            // Fallback general si hay error de red
            if (scGeoLat === null || isNaN(scGeoLat)) {
                scGeoLat = -34.6037;
                scGeoLng = -58.3816;
                scShowMiniMap(scGeoLat, scGeoLng);
            }
        }
        if (loader) loader.style.display = 'none';
    }, 1000);
}

function scShowMiniMap(lat, lng) {
    var wrap = document.getElementById('sc-mini-map-wrap');
    wrap.style.display = 'block';
    if (!scMiniMap) {
        scMiniMap = L.map('sc-mini-map', { zoomControl: false, attributionControl: false }).setView([lat, lng], 16);
        L.tileLayer('https://{s}.basemaps.cartocdn.com/rastertiles/voyager/{z}/{x}/{y}{r}.png').addTo(scMiniMap);
        scMiniMarker = L.marker([lat, lng], { draggable: true }).addTo(scMiniMap);
        scMiniMarker.on('dragend', function() {
            var p = scMiniMarker.getLatLng();
            scGeoLat = p.lat; scGeoLng = p.lng;
        });
    } else {
        scMiniMap.setView([lat, lng], 16);
        scMiniMarker.setLatLng([lat, lng]);
    }
}

async function scSubmitAlerta() {
    var pais = document.getElementById('sc-pais').value;
    var ciudad = document.getElementById('sc-ciudad').value.trim();
    var dir = document.getElementById('sc-direccion').value.trim();
    var obs = document.getElementById('sc-observaciones').value.trim();
    
    if (!dir) {
        document.getElementById('sc-direccion').focus();
        document.getElementById('sc-direccion').style.borderColor = '#e74c3c';
        return;
    }
    
    if (scGeoLat === null || scGeoLng === null || isNaN(scGeoLat) || isNaN(scGeoLng)) {
        // Fallback silencioso a Buenos Aires si fallÃƒÂ³ la carga del mapa o el dato es invÃƒÂ¡lido
        scGeoLat = -34.6037;
        scGeoLng = -58.3816;
    }
    
    var direccionCompleta = dir + (ciudad ? ', ' + ciudad : '') + ' (' + pais + ')';
    
    var btn = document.getElementById('sc-submit-btn');
    btn.disabled = true;
    btn.innerHTML = '<i class="ri-loader-4-line ri-spin"></i> Publicando...';

    var u = (typeof auth !== 'undefined' && auth) ? auth.getCurrentUser() : null;
    var isValidUUID = u && u.id && /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(u.id);
    var payload = {
        direccion: direccionCompleta,
        necesidades: scNecesidades.join(','),
        descripcion: obs,
        lat: scGeoLat,
        lng: scGeoLng
    };
    if (scFotoBase64) payload.foto_url = scFotoBase64;
    
    try {
        if (typeof sbClient !== 'undefined' && sbClient) {
            var error = null;
            if (scEditingId) {
                var { error: errUpdate } = await sbClient.from('alertas_calle').update(payload).eq('id', scEditingId);
                error = errUpdate;
            } else {
                payload.user_id = isValidUUID ? u.id : null;
                payload.reporter_name = u ? (u.name || (u.email ? u.email.split('@')[0] : 'AnÃƒÂ³nimo')) : 'AnÃƒÂ³nimo';
                payload.activa = true;
                payload.created_at = new Date().toISOString();
                var { error: errInsert } = await sbClient.from('alertas_calle').insert([payload]);
                error = errInsert;
            }
            
            if (error) {
                btn.innerHTML = '<i class="ri-error-warning-line"></i> ' + (error.message || 'Error desconocido');
                btn.disabled = false;
                return; // Stop execution
            }
        }
        scEditingId = null;
        scCloseForm();
        await scLoadAlertas();
        if (scGeoLat && scGeoLng) scFocusAlerta(scGeoLat, scGeoLng);
        scCurrentTab = 'todas';
        scChangeTab('todas');
        btn.disabled = false;
        btn.innerHTML = '<i class="ri-send-plane-fill"></i> Publicar Alerta';
        document.getElementById('sc-observaciones').value = '';
        document.getElementById('sc-direccion').value = '';
        document.getElementById('sc-ciudad').value = '';
        document.getElementById('sc-mini-map-wrap').style.display = 'none';
        scNecesidades = [];
        document.querySelectorAll('.sc-nec-chip').forEach(function(c) { c.classList.remove('active'); });
        if (typeof scRemoveFoto === 'function') scRemoveFoto();
    } catch(e) {
        btn.innerHTML = '<i class="ri-error-warning-line"></i> Error JS';
        btn.disabled = false;
        console.error("Submit Alerta Error:", e);
    }
}

var scFotoBase64 = null;

function scHandleFotoUpload(input) {
    if (input.files && input.files[0]) {
        var file = input.files[0];
        if (file.size > 8 * 1024 * 1024) { alert('La foto es demasiado grande. Intenta con una menor a 8MB.'); return; }
        
        var reader = new FileReader();
        reader.onload = function(e) {
            var img = new Image();
            img.onload = function() {
                var canvas = document.createElement('canvas');
                var maxW = 600; var maxH = 600;
                var w = img.width; var h = img.height;
                if (w > h && w > maxW) { h = Math.round((h * maxW) / w); w = maxW; }
                else if (h > maxW) { w = Math.round((w * maxH) / h); h = maxH; }
                canvas.width = w; canvas.height = h;
                var ctx = canvas.getContext('2d');
                ctx.drawImage(img, 0, 0, w, h);
                scFotoBase64 = canvas.toDataURL('image/jpeg', 0.6); // Compress heavy
                
                document.getElementById('sc-foto-img').src = scFotoBase64;
                document.getElementById('sc-foto-preview').style.display = 'block';
                document.getElementById('sc-foto-label').style.display = 'none';
            };
            img.src = e.target.result;
        };
        reader.readAsDataURL(file);
    }
}

function scRemoveFoto() {
    scFotoBase64 = null;
    var input = document.getElementById('sc-foto-input');
    if (input) input.value = '';
    var preview = document.getElementById('sc-foto-preview');
    if (preview) preview.style.display = 'none';
    var label = document.getElementById('sc-foto-label');
    if (label) label.style.display = 'flex';
}

function scEscape(s) {
    if (!s) return '';
    return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
